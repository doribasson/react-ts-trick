let a=5;
let b=a;
b++;
ה b עכשיו 6 אבל ה a ישאר 5.

כשמדובר על אובייקט מערך או קלס
ההשמה by reference זאת אומרת שהכתובת משתנה ולא התוכן.
const a = [1,2];
let b= a;

a.push(6);

עכשיו ה a יהיה 6 ו b יהיה 6 גם כי בעצם  הלכנו לאותו כתובת של a ו b  ומשם הלכמו ושינינו את התוכן. זה בי רפרנס כי שיננו כתובת.


 [1,2] === [1,2] //false

שונים כי זה רפרנסים שונים למרות שהתוכן זהה. לכל אחד יש כתובת שונה


כשמעבירים מערך או אובייקט לפונקציה זה מועבר בי רפרנס זאת אומרת שמקבלים את הכתובת שלו ככה שהוא משתנה. אם בתוך הפונקציה נעשה השמה =array הוא ישנה את הכתובת ולכן השינוי של התוכן לא יתבצע אלא רק הכתובת שלו זאת אומרת ילך למקום אחר בזיכרון וינתק את עצמו מהכתובת שהייתה לו כשקיבל בפונקציה והשינוי לא יתבצע. לכן בשהיל שינוי נצטרך או לעשות העתקה עמוקה עם ספרידאופרטור או להשתמש ב array.push
